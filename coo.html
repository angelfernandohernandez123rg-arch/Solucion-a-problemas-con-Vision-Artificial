üíª C√≥digo del Ejemplo

Aprende a recortar secciones espec√≠ficas de una imagen usando coordenadas. Las im√°genes en OpenCV son matrices, por lo que puedes acceder a regiones usando notaci√≥n de slicing [y1:y2, x1:x2]. 

def recortes():
    import cv2
    # Importa librer√≠a opencv

    # Cargar imagen
    gato = cv2.imread('GATO.jpg')
    
    while True:
        recorte = gato[20:600, 50:600]  # alto y ancho

        cv2.imshow('imagen original', gato)
        cv2.imshow('imagen recortada', recorte)

        tecla = cv2.waitKey(0)

        if tecla == ord('x'):
            break

    cv2.destroyAllWindows()

# __main__ es el nombre interno del archivo cuando se ejecute en s√≠ mismo
if __name__ == '__main__':
    recortes()
