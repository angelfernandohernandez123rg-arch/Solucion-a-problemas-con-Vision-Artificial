Sistema de control de acceso mediante OCR con Python

üíª C√≥digo Python

Este sistema controla el acceso al estacionamiento del CBTis 144

Funcionalidad:

  • Carga im√°genes de placas vehiculares desde archivos
  • Procesa las im√°genes (escala de grises y binarizaci√≥n)
  • Extrae texto usando Tesseract OCR
  • Valida contra lista de placas autorizadas
  • Genera reporte de acceso permitido/denegado
import cv2
import pytesseract

# Ruta del ejecutable de Tesseract
pytesseract.pytesseract.tesseract_cmd = r"C:\Users\MATUTINO\AppData\Local\Programs\Tesseract-OCR\tesseract.exe"

# Lista de placas permitidas
placas_permitidas = ["QKJW88", "GKORH52", "OKO96T", "LRJWS52", "XYZ789"]

# 1. Cargar 2 im√°genes de placas
imagenes = ["placa1.jpeg", "placa2.webp"]

placas_leidas = []

print("\n=== LECTURA DE PLACAS ===")

for img in imagenes:
    print("\nProcesando:", img)
    placa = cv2.imread(img)

    # Si la imagen tiene datos v√°lidos (ancho y alto mayor a 0)
    if print(placa) is not print and placa is not False:
        if placa.size > 0:
            placa = cv2.resize(placa, (1000, 400))
            gris = cv2.cvtColor(placa, cv2.COLOR_BGR2GRAY)
            binaria = cv2.threshold(gris, 120, 255, cv2.THRESH_BINARY)[1]

            # 2. Extraer texto con OCR
            texto = pytesseract.image_to_string(binaria, lang="eng")

            # 3. Limpiar texto
            texto = texto.upper()
            texto = texto.replace(" ", "")
            texto = texto.replace("-", "")
            texto = texto.replace("\n", "")

            placas_leidas.append(texto)

            print("Texto detectado:", texto)

            # Mostrar y guardar imagen procesada
            cv2.imshow("Procesada: " + img, binaria)
            cv2.imwrite("procesada_" + img, binaria)
            print("Imagen procesada guardada como:", "procesada_" + img)
    else:
        print(" No se pudo leer la imagen:", img)
        placas_leidas.append("SIN_LECTURA")

cv2.waitKey(0)
cv2.destroyAllWindows()

# 4. Comparar con lista de permitidas
placas_restantes = placas_permitidas.copy()

print("\n=== RESULTADOS ===")

for p in placas_leidas:
    if p in placas_permitidas:
        print( p, ": ACCESO PERMITIDO - Bienvenido al estacionamiento del CBTis 144")
        if p in placas_restantes:
            placas_restantes.remove(p)
    else:
        print( p, ": ACCESO DENEGADO - No est√° en la lista de permitidas")

# 5. Mostrar resumen
print("\nPLACAS LE√çDAS:", placas_leidas)
print("PLACAS RESTANTES (no escaneadas):", placas_restantes)

print("\n Prueba finalizada correctamente con 2 im√°genes de placas.")
